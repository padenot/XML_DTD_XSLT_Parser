\newcommand{\partfont}{
	\fontencoding{T1}
	\fontfamily{phv}
	\fontseries{bc}
	\fontshape{n}
	\fontsize{32}{30}
	\selectfont
}

\DeclareFixedFont{\partnumfont}{T1}{phv}{bc}{n}{80}

\makeatletter
\renewcommand\part
{
	\let\@font@warningORIG\@font@warning
	\renewcommand{\@font@warning}[1]{}
	\cleardoublepage
	\thispagestyle{empty}
	\@tempswafalse
	\null\vfil
	\secdef\@part\@spart
}


\def\@part[#1]#2{
	\ifnum \c@secnumdepth >-2\relax
		\refstepcounter{part}
		\addcontentsline{toc}{part}{\thepart\hspace{1em}#1}
	\else
		\addcontentsline{toc}{part}{#1}
	\fi 
	\markboth{}{}
	{
	\centering
	\interlinepenalty \@M 
	\normalfont
	\ifnum \c@secnumdepth >-2\relax
		\partnumfont \thepart
		\par
		\vskip 50\p@ 
	\fi 
	\partfont #2\par} 
	\@endpart
	\let\@font@warning\@font@warningORIG
}

\makeatother
